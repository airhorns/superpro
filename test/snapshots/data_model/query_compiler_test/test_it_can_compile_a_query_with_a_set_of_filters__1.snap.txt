SELECT SUM("warehouse"."fct_shopify_orders"."total_price") AS sales__orderfacts__total_price__sum, date_trunc('year', "warehouse"."fct_shopify_orders"."created_at") AS sales__orderfacts__created_at__date_trunc_year FROM "warehouse"."fct_shopify_orders" WHERE (date_trunc('year', "warehouse"."fct_shopify_orders"."created_at") AS sales__orderfacts__created_at__date_trunc_year = '2000') AND (SUM("warehouse"."fct_shopify_orders"."total_price") AS sales__orderfacts__total_price__sum > 100) AND "warehouse"."fct_shopify_orders"."account_id" = 10 GROUP BY sales__orderfacts__created_at__date_trunc_year ORDER BY sales__orderfacts__created_at__date_trunc_year ASC LIMIT 5000
