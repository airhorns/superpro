SELECT COUNT(case when "warehouse"."fct_shopify_repurchase_intervals"."order_seq_number" = 1 and "warehouse"."fct_shopify_repurchase_intervals"."days_until_next_order" <= 60.0 then 1 end) * 1.0 / nullif(COUNT(*), 0) AS sales__repurchaseintervalfacts__early_repurchase_rate, date_trunc('day', "warehouse"."fct_shopify_repurchase_intervals"."order_date") AS sales__repurchaseintervalfacts__order_date__date_trunc_day FROM "warehouse"."fct_shopify_repurchase_intervals" WHERE "warehouse"."fct_shopify_repurchase_intervals"."account_id" = 10 GROUP BY sales__repurchaseintervalfacts__order_date__date_trunc_day ORDER BY sales__repurchaseintervalfacts__order_date__date_trunc_day ASC LIMIT 5000
