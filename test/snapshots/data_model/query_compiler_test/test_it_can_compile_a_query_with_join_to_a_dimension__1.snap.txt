SELECT SUM("warehouse"."fct_shopify_orders"."total_price") AS sales__orderfacts__total_price__sum, "warehouse"."dim_shopify_customers"."first_name" AS sales__orderfacts__customer_first_name FROM "warehouse"."fct_shopify_orders" LEFT OUTER JOIN "warehouse"."dim_shopify_customers" ON "warehouse"."fct_shopify_orders"."customer_id" = "warehouse"."dim_shopify_customers"."customer_id" WHERE "warehouse"."fct_shopify_orders"."account_id" = 10 GROUP BY sales__orderfacts__customer_first_name ORDER BY sales__orderfacts__total_price__sum DESC LIMIT 5000
