SELECT SUM("warehouse"."fct_shopify_orders"."total_price") AS sales__orderfacts__total_price__sum, AVG("warehouse"."fct_shopify_orders"."total_weight") AS sales__orderfacts__total_weight__average, "warehouse"."fct_shopify_orders"."cancelled" AS sales__orderfacts__cancelled FROM "warehouse"."fct_shopify_orders" WHERE "warehouse"."fct_shopify_orders"."account_id" = 10 GROUP BY sales__orderfacts__cancelled ORDER BY sales__orderfacts__total_price__sum DESC LIMIT 5000
