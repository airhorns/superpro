SELECT SUM("warehouse"."fct_shopify_orders"."total_price") AS sales__orderfacts__total_price__sum, date_trunc('day', "warehouse"."fct_shopify_orders"."created_at") AS sales__orderfacts__created_at__date_trunc_day FROM "warehouse"."fct_shopify_orders" WHERE "warehouse"."fct_shopify_orders"."account_id" = 10 GROUP BY sales__orderfacts__created_at__date_trunc_day ORDER BY sales__orderfacts__created_at__date_trunc_day ASC LIMIT 5000
